{%- comment -%} Renders a single VEVENT block for `include.event`. Shared by the per-event .ics files and the combined subscription feed (/events/all.ics). Escapes text per RFC 5545 (backslash, semicolon, comma, newline). {%- endcomment -%} {%- assign ev = include.event -%} {%- assign summary = ev.calendar_title | default: ev.title -%} {%- assign descsrc = ev.calendar_description | default: ev.lede | default: "" -%} {%- assign NL = " " -%} {%- assign summary_e = summary | replace: "\", "\\" | replace: ";", "\;" | replace: ",", "\," -%} {%- assign loc_e = ev.location | default: "" | replace: "\", "\\" | replace: ";", "\;" | replace: ",", "\," -%} {%- capture descfull -%}{{ descsrc }}{%- if ev.info_url %}{{ NL }}{{ NL }}{{ ev.info_url }}{%- endif -%}{%- endcapture -%} {%- assign desc_e = descfull | replace: "\", "\\" | replace: ";", "\;" | replace: ",", "\," | replace: NL, "\n" -%} {%- assign d = ev.date | date: "%Y%m%d" -%} {%- assign ts = ev.start | default: "00:00" | replace: ":", "" -%} {%- assign te = ev.end | default: ev.start | default: "00:00" | replace: ":", "" -%} BEGIN:VEVENT UID:{{ ev.ref }}@salsalastyle.dk DTSTAMP:20260101T000000Z DTSTART;TZID=Europe/Copenhagen:{{ d }}T{{ ts }}00 DTEND;TZID=Europe/Copenhagen:{{ d }}T{{ te }}00 SUMMARY:{{ summary_e }} {% if ev.location %}LOCATION:{{ loc_e }} {% endif %}{% if desc_e != "" %}DESCRIPTION:{{ desc_e }} {% endif %}{% if ev.info_url %}URL:{{ ev.info_url }} {% endif %}END:VEVENT
